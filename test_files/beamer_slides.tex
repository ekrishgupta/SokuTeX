\section{Introduction}

\begin{frame}{What is SokuTeX?}
\begin{itemize}
    \item A native LaTeX editor written in Rust
    \item GPU-accelerated PDF preview with wgpu
    \item Real-time compilation with multiple backends
    \item SyncTeX support for bidirectional navigation
\end{itemize}
\end{frame}

\section{Features}

\begin{frame}{Architecture}
\begin{block}{Compilation Pipeline}
    Editor $\to$ VFS $\to$ Compiler $\to$ PDF Renderer $\to$ GPU
\end{block}

\begin{alertblock}{Performance}
    Near-instant feedback via shadow rendering
\end{alertblock}

\begin{exampleblock}{Math Support}
    Full $\LaTeX$ math: $\int_0^\infty e^{-x^2} dx = \frac{\sqrt{\pi}}{2}$
\end{exampleblock}
\end{frame}

\begin{frame}{Incremental Compilation}
\begin{columns}
\column{0.5\textwidth}
    \textbf{Traditional}
    \begin{enumerate}
        \item Edit file
        \item Recompile everything
        \item Wait 5-10s
        \item View PDF
    \end{enumerate}

\column{0.5\textwidth}
    \textbf{SokuTeX}
    \begin{enumerate}
        \item Edit file
        \item Detect changed subtree
        \item Recompile \textit{only} changes
        \item View in $<$150ms
    \end{enumerate}
\end{columns}
\end{frame}
