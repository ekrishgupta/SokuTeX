\section{Nested Environments Stress}
\begin{theorem}
For all $\epsilon > 0$, there exists $\delta > 0$ such that
\[
    |x - a| < \delta \implies |f(x) - f(a)| < \epsilon
\]
\end{theorem}

\begin{proof}
Choose $\delta = \min\left\{1, \frac{\epsilon}{2M}\right\}$ where $M = \sup_{|x-a| \leq 1} |f'(x)|$.
Then by the mean value theorem:
\begin{align*}
    |f(x) - f(a)| &= |f'(\xi)||x - a| \\
                   &\leq M \cdot \delta \\
                   &\leq M \cdot \frac{\epsilon}{2M} \\
                   &= \frac{\epsilon}{2} < \epsilon. \qedhere
\end{align*}
\end{proof}
